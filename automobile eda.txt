# ==========================================================
# AUTOMOBILE DATASET - EXPLORATORY DATA ANALYSIS (EDA)
# File Name : Automobile_data.csv
# ==========================================================

# Import Libraries
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

# ----------------------------------------------------------
# Read Dataset
# ----------------------------------------------------------
df = pd.read_csv("Automobile_data.csv")

# ----------------------------------------------------------
# Display Dataset
# ----------------------------------------------------------
print("First 5 Rows")
print(df.head())

print("\nLast 5 Rows")
print(df.tail())

print("\nShape of Dataset")
print(df.shape)

print("\nColumns")
print(df.columns)

print("\nInformation")
print(df.info())

print("\nStatistical Summary")
print(df.describe())

# ----------------------------------------------------------
# Missing Values
# ----------------------------------------------------------
df.replace("?", np.nan, inplace=True)

print("\nMissing Values")
print(df.isnull().sum())

# ----------------------------------------------------------
# Convert Numeric Columns
# ----------------------------------------------------------
num_cols = ['normalized-losses',
            'bore',
            'stroke',
            'horsepower',
            'peak-rpm',
            'price']

for col in num_cols:
    if col in df.columns:
        df[col] = pd.to_numeric(df[col], errors='coerce')

# Fill Missing Values
for col in num_cols:
    if col in df.columns:
        df[col].fillna(df[col].median(), inplace=True)

print("\nMissing Values After Cleaning")
print(df.isnull().sum())

# ----------------------------------------------------------
# Duplicate Rows
# ----------------------------------------------------------
print("\nDuplicate Rows =", df.duplicated().sum())

# ----------------------------------------------------------
# Correlation Matrix
# ----------------------------------------------------------
print("\nCorrelation Matrix")
print(df.corr(numeric_only=True))

# ==========================================================
# DATA VISUALIZATION
# ==========================================================

sns.set(style="whitegrid")

# ----------------------------------------------------------
# Histogram
# ----------------------------------------------------------
plt.figure(figsize=(8,5))
sns.histplot(df['price'], bins=20, kde=True, color='skyblue')
plt.title("Distribution of Car Prices")
plt.xlabel("Price")
plt.ylabel("Frequency")
plt.show()

# ----------------------------------------------------------
# Count Plot
# ----------------------------------------------------------
plt.figure(figsize=(6,5))
sns.countplot(x='fuel-type', data=df)
plt.title("Fuel Type Count")
plt.show()

# ----------------------------------------------------------
# Box Plot
# ----------------------------------------------------------
plt.figure(figsize=(8,5))
sns.boxplot(x='fuel-type', y='price', data=df)
plt.title("Fuel Type vs Price")
plt.show()

# ----------------------------------------------------------
# Scatter Plot
# ----------------------------------------------------------
plt.figure(figsize=(8,5))
sns.scatterplot(x='horsepower', y='price', data=df)
plt.title("Horsepower vs Price")
plt.show()

# ----------------------------------------------------------
# Bar Plot
# ----------------------------------------------------------
plt.figure(figsize=(12,6))
sns.barplot(x='make', y='price', data=df)
plt.xticks(rotation=90)
plt.title("Average Price by Car Company")
plt.show()

# ----------------------------------------------------------
# Pie Chart
# ----------------------------------------------------------
fuel = df['fuel-type'].value_counts()

plt.figure(figsize=(6,6))
plt.pie(fuel,
        labels=fuel.index,
        autopct="%1.1f%%",
        startangle=90)

plt.title("Fuel Type Distribution")
plt.show()

# ----------------------------------------------------------
# Violin Plot
# ----------------------------------------------------------
plt.figure(figsize=(8,5))
sns.violinplot(x='fuel-type', y='price', data=df)
plt.title("Price Distribution by Fuel Type")
plt.show()

# ----------------------------------------------------------
# Line Plot
# ----------------------------------------------------------
sorted_df = df.sort_values("engine-size")

plt.figure(figsize=(8,5))
plt.plot(sorted_df["engine-size"],
         sorted_df["price"],
         marker='o')

plt.title("Engine Size vs Price")
plt.xlabel("Engine Size")
plt.ylabel("Price")
plt.show()

# ----------------------------------------------------------
# Pair Plot
# ----------------------------------------------------------
sns.pairplot(df[['price',
                 'horsepower',
                 'engine-size',
                 'city-mpg']])
plt.show()

# ----------------------------------------------------------
# Heatmap
# ----------------------------------------------------------
plt.figure(figsize=(12,8))
sns.heatmap(df.corr(numeric_only=True),
            annot=True,
            cmap="coolwarm",
            fmt=".2f")

plt.title("Correlation Heatmap")
plt.show()

# ==========================================================
# Additional EDA
# ==========================================================

print("\nTop 10 Most Expensive Cars")
print(df[['make','price']].sort_values(by='price',
                                       ascending=False).head(10))

print("\nAverage Price by Company")
print(df.groupby('make')['price'].mean())

print("\nAverage Horsepower by Company")
print(df.groupby('make')['horsepower'].mean())

print("\nFuel Type Count")
print(df['fuel-type'].value_counts())

print("\nBody Style Count")
print(df['body-style'].value_counts())

print("\nDrive Wheels Count")
print(df['drive-wheels'].value_counts())

print("\nNumber of Cars by Company")
print(df['make'].value_counts())

print("\nEDA Completed Successfully!")
